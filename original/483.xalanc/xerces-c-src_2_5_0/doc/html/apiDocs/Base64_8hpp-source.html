00244 inline void Base64::split1stOctet(const XMLByte& ch, XMLByte& b1, XMLByte& b2) {
00245     b1 = ch >> 2;
00246     b2 = ( ch & 0x3 ) << 4;
00247 }

00249 inline void Base64::split2ndOctet(const XMLByte& ch, XMLByte& b2, XMLByte& b3) {
00250     b2 |= ch >> 4;  // combine with previous value
00251     b3 = ( ch & 0xf ) << 2;
00252 }
00253 
00254 inline void Base64::split3rdOctet(const XMLByte& ch, XMLByte& b3, XMLByte& b4) {
00255     b3 |= ch >> 6;  // combine with previous value
00256     b4 = ( ch & 0x3f );
